% Generic logo helper: include the image if it exists, otherwise show text.
% Use this for technologies where you may add a logo later.
\newcommand{\LogoOrText}[2]{%
  \IfFileExists{#1}{%
    \begin{minipage}[c][1.95cm][c]{\linewidth}
      \centering
      \includegraphics[width=0.78\linewidth,height=1.55cm,keepaspectratio]{#1}
    \end{minipage}%
  }{%
    \begin{minipage}[c][1.95cm][c]{\linewidth}
      \centering
      \textbf{#2}
    \end{minipage}%
  }%
}
